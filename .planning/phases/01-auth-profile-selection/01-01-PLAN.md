2. Rewrite `src/app/services/auth.service.ts` completely. The new service:

   ```typescript
   import { Injectable } from '@angular/core';
   import { HttpClient } from '@angular/common/http';
   import { BehaviorSubject, Observable, firstValueFrom } from 'rxjs';
   import { HubUser, Allowlist } from '../models/user.model';
   import { initializeApp, FirebaseApp } from 'firebase/app';
   import {
     getAuth,
     signInWithPopup,
     GoogleAuthProvider,
     signOut,
     Auth,
     onAuthStateChanged,
     UserCredential,
   } from 'firebase/auth';
   import { environment } from '../../environments/environment';

   const STORAGE_KEY = 'dropi_hub_user';

   @Injectable({ providedIn: 'root' })
   export class AuthService {
     private firebaseApp: FirebaseApp;
     private auth: Auth;
     private provider = new GoogleAuthProvider();
     private currentUser$ = new BehaviorSubject<HubUser | null>(this.loadFromStorage());
     private _authError$ = new BehaviorSubject<string | null>(null);

     /** Emits current authenticated user or null */
     readonly user$: Observable<HubUser | null> = this.currentUser$.asObservable();
     /** Emits auth error message (e.g. unauthorized) */
     readonly authError$: Observable<string | null> = this._authError$.asObservable();

     get currentUser(): HubUser | null {
       return this.currentUser$.value;
     }

     get isAuthenticated(): boolean {
       return this.currentUser$.value !== null;
     }

     constructor(private http: HttpClient) {
       this.firebaseApp = initializeApp(environment.firebaseConfig);
       this.auth = getAuth(this.firebaseApp);

       // Sync Firebase auth state on page load
       onAuthStateChanged(this.auth, (fbUser) => {
         if (fbUser && !this.currentUser$.value) {
           // Firebase has a session but localStorage was cleared — restore
           const hubUser: HubUser = {
             uid: fbUser.uid,
             email: fbUser.email ?? '',
             displayName: fbUser.displayName ?? '',
             photoURL: fbUser.photoURL,
             createdAt: new Date().toISOString(),
           };
           this.setUser(hubUser);
         }
       });
     }

     /**
      * Sign in with Google popup. Returns the HubUser on success.
      * Throws error string if email is not in allowlist.
      */
     async loginWithGoogle(): Promise<HubUser> {
       this._authError$.next(null);

       try {
         const result: UserCredential = await signInWithPopup(this.auth, this.provider);
         const email = result.user.email ?? '';

         // Check allowlist
         const allowed = await this.isEmailAllowed(email);
         if (!allowed) {
           // Sign out from Firebase immediately
           await signOut(this.auth);
           const errorMsg = 'Acceso no autorizado — contacta a producto@dropi.co';
           this._authError$.next(errorMsg);
           throw new Error(errorMsg);
         }

         const hubUser: HubUser = {
           uid: result.user.uid,
           email,
           displayName: result.user.displayName ?? '',
           photoURL: result.user.photoURL,
           createdAt: new Date().toISOString(),
         };

         this.setUser(hubUser);
         return hubUser;
       } catch (err: any) {
         // Re-throw allowlist errors as-is
         if (err?.message?.includes('Acceso no autorizado')) {
           throw err;
         }
         // Firebase popup errors (user closed, network, etc.)
         const errorMsg = 'Error al iniciar sesión con Google. Inténtalo de nuevo.';
         this._authError$.next(errorMsg);
         throw new Error(errorMsg);
       }
     }

     async logout(): Promise<void> {
       localStorage.removeItem(STORAGE_KEY);
       await signOut(this.auth);
       this.currentUser$.next(null);
       this._authError$.next(null);
     }

     /**
      * Check if email is in the allowlist:
      * 1. Domain matches any in allowlist.domains -> approved
      * 2. Email matches any in allowlist.emails -> approved
      * 3. Otherwise -> rejected
      */
     private async isEmailAllowed(email: string): Promise<boolean> {
       try {
         const allowlist = await firstValueFrom(
           this.http.get<Allowlist>('/allowed-emails.json')
         );
         const emailLower = email.toLowerCase();
         const domain = emailLower.split('@')[1];

         // Check domain first
         if (allowlist.domains?.some(d => d.toLowerCase() === domain)) {
           return true;
         }
         // Check individual emails
         if (allowlist.emails?.some(e => e.toLowerCase() === emailLower)) {
           return true;
         }
         return false;
       } catch {
         // If allowlist fetch fails, deny access for safety
         return false;
       }
     }

     private setUser(user: HubUser): void {
       localStorage.setItem(STORAGE_KEY, JSON.stringify(user));
       this.currentUser$.next(user);
     }

     private loadFromStorage(): HubUser | null {
       const raw = localStorage.getItem(STORAGE_KEY);
       if (raw) {
         try {
           return JSON.parse(raw) as HubUser;
         } catch {
           return null;
         }
       }
       return null;
     }
   }
   ```

   Key design decisions:
   - Firebase is initialized in the AuthService constructor (not via APP_INITIALIZER) to keep it simple and self-contained.
   - The allowlist is fetched via HttpClient from the app's served assets (not imported as a module — this way it can be updated without rebuilding).
   - `signInWithPopup` is used (not redirect) for simpler UX flow.
   - On unauthorized email: Firebase session is immediately signed out so the user cannot bypass the allowlist.
   - The `authError$` observable lets the login component reactively display errors.
   - The old `login()` and `loginAs()` methods are REMOVED — they were mock-only.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Google SSO -> App | Firebase returns user email; app must verify against allowlist |
| allowed-emails.json -> AuthService | JSON file served as static asset; anyone with URL can read it |
| localStorage -> App | Stored session could be tampered with in browser devtools |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-01-01 | Spoofing | AuthService.loginWithGoogle | mitigate | Firebase Auth verifies Google identity; app checks email against allowlist AFTER Firebase auth |
| T-01-02 | Tampering | localStorage session | accept | Client-side only app; user can tamper with own session but cannot access others' data. No sensitive data beyond email/name. |
| T-01-03 | Information Disclosure | allowed-emails.json | accept | File contains only email addresses of authorized users, not credentials. Acceptable for internal tool. |
| T-01-04 | Elevation of Privilege | Allowlist bypass | mitigate | Unauthorized email triggers immediate `signOut(this.auth)` before storing any session data |
</threat_model>
